import requests

def fetch_nws_climate_report(url):
    r = requests.get(url, timeout=20)
    r.raise_for_status()
    return r.text

import requests

def fetch_ncei_station_data(url, params=None):
    r = requests.get(url, params=params or {}, timeout=20)
    r.raise_for_status()
    return r.json()
