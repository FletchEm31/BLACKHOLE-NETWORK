def fetch_and_normalize_catalog(db_config):
    """
    Queries weather_kalshi_contract_catalog, parses threshold vs interval markets,
    and isolates clean floating point strike bounds.
    """
    conn = psycopg2.connect(**db_config, cursor_factory=RealDictCursor)
    cursor = conn.cursor()
    
    query = """
        SELECT ticker, station_code, strike_type, numeric_strike_low, numeric_strike_high, raw_json
        FROM weather_kalshi_contract_catalog 
        WHERE is_active = true;
    """
    cursor.execute(query)
    raw_rows = cursor.fetchall()
    
    normalized_contracts = []
    for row in raw_rows:
        ticker = row['ticker']
        s_type = row['strike_type']
        low_val = float(row['numeric_strike_low'])
        high_val = float(row['numeric_strike_high'])
        
        # Guard against raw unnested threshold bugs (e.g. T68 ceiling bounds)
        if s_type == "threshold" or row['raw_json'].get('strike_type') == "less":
            if "-T" in ticker:
                strike_low_clean = -999.0
                strike_high_clean = high_val - 1.0  # T68 means 67.0°F or below
            else:
                strike_low_clean = low_val          # B88 means 88.0°F or above
                strike_high_clean = 999.0
        else:
            strike_low_clean = low_val
            strike_high_clean = high_val
            
        normalized_contracts.append({
            'ticker': ticker,
            'station_code': row['station_code'],
            'strike_low': strike_low_clean,
            'strike_high': strike_high_clean,
            'yes_bid': float(row['raw_json'].get('yes_bid_dollars', 0.0)) * 100,
            'yes_ask': float(row['raw_json'].get('yes_ask_dollars', 1.0)) * 100,
            'no_bid': float(row['raw_json'].get('no_bid_dollars', 0.0)) * 100,
            'no_ask': float(row['raw_json'].get('no_ask_dollars', 1.0)) * 100,
            'is_illiquid': row['raw_json'].get('status', 'active') == 'illiquid' or 
                           float(row['raw_json'].get('liquidity_dollars', 0)) < 500
        })
        
    cursor.close()
    conn.close()
    return pd.DataFrame(normalized_contracts)

def fetch_live_features_and_sigmas(db_config):
    """
    Extracts engineered features for running models and calculates the trailing
    30-day RMSE from the silver layer to drive dynamic probability density functions.
    """
    conn = psycopg2.connect(**db_config, cursor_factory=RealDictCursor)
    cursor = conn.cursor()
    
    # Extract dynamic rolling standard error to prevent overconfidence spikes
    sigma_query = """
        WITH daily_errors AS (
            SELECT location, POWER(corrected_forecast_temp - actual_high_temp, 2) as se
            FROM weather_silver_forecast_error 
            WHERE forecast_date >= CURRENT_DATE - INTERVAL '30 days'
        )
        SELECT location, SQRT(AVG(se)) as current_rmse FROM daily_errors GROUP BY location;
    """
    cursor.execute(sigma_query)
    sigmas = {row['location']: float(row['current_rmse']) for row in cursor.fetchall()}
    
    # Pull current active features from Gold layer
    feature_query = """
        SELECT station_code, latest_nws_tmax_f, latest_gfs_tmax_f, dewpoint_spread, 
               humidity_spread, cloud_cover_change, wind_shift_flag
        FROM weather_gold_city_day_features;
    """
    feature_df = pd.read_sql(feature_query, conn)
    
    cursor.close()
    conn.close()
    return feature_df, sigmas

def generate_bias_corrected_forecast(feature_df, model_path="weather_silver_model_base.json"):
    """
    Loads pretrained XGBoost model, applies inference on live features,
    and returns localized forecast outputs.
    """
    # Initialize XGBoost regressor framework
    bst = xgb.XGBRegressor()
    bst.load_model(model_path)
    
    # One-hot encoding preparation to match structural training baselines
    encoded_df = pd.get_dummies(feature_df, columns=['station_code'], drop_first=False)
    
    # Ensure standard evaluation matrices are aligned
    features = encoded_df.drop(columns=['latest_nws_tmax_f'], errors='ignore')
    predictions = bst.predict(features)
    
    feature_df['bias_corrected_mean'] = predictions
    return feature_df

def run_sweeper_engine(contracts_df, feature_df, sigmas, total_bankroll):
    """
    Combines weather parameters and book states to generate edge calculations,
    allocates positions via Fractional Kelly, and flags trade signals.
    """
    # Merge catalog frameworks with model calculations
    merged = contracts_df.merge(feature_df, on='station_code', how='inner')
    
    trading_signals = []
    for idx, row in merged.iterrows():
        station = row['station_code']
        pred_mean = row['bias_corrected_mean']
        
        # Fetch localized dynamic standard deviation with a firm fallback floor
        sigma = max(sigmas.get(station, 1.35), 0.45)
        
        # Execute integration across the Gaussian Cumulative Distribution Function
        p_below_high = stats.norm.cdf(row['strike_high'], loc=pred_mean, scale=sigma)
        p_below_low = stats.norm.cdf(row['strike_low'], loc=pred_mean, scale=sigma)
        fair_value_prob = p_below_high - p_below_low
        fair_value_cents = int(fair_value_prob * 100)
        
        # Determine book boundaries and premium friction thresholds
        min_edge = 8 if row['is_illiquid'] else 5
        yes_ask = row['yes_ask']
        yes_bid = row['yes_bid']
        
        action = "SKIP"
        calculated_edge = 0
        target_price = 0
        
        # Strategy A: Market underpricing -> Buy YES contracts
        if yes_ask > 0 and (fair_value_cents - yes_ask >= min_edge):
            action = "BUY_YES"
            calculated_edge = fair_value_cents - yes_ask
            target_price = yes_ask
            market_odds = yes_ask / 100.0
            
        # Strategy B: Market overpricing -> Buy NO contracts
        elif yes_bid > 0 and (yes_bid - fair_value_cents >= min_edge):
            action = "BUY_NO"
            calculated_edge = yes_bid - fair_value_cents
            target_price = 100 - yes_bid
            market_odds = target_price / 100.0
            
        # Sizing loop execution via Quarter-Kelly Allocation
        contract_count = 0
        if action != "SKIP" and market_odds > 0:
            b = (1.0 - market_odds) / market_odds
            p = fair_value_prob if action == "BUY_YES" else (1.0 - fair_value_prob)
            q = 1.0 - p
            
            kelly_f = (p * b - q) / b
            quarter_kelly = max(0, kelly_f * 0.25)
            
            allocated_capital = total_bankroll * quarter_kelly
            contract_count = int(allocated_capital / (target_price / 100.0)) if target_price > 0 else 0

        if action != "SKIP" and contract_count > 0:
            trading_signals.append({
                'ticker': row['ticker'],
                'station': station,
                'action': action,
                'limit_price_cents': target_price,
                'size': contract_count,
                'computed_edge_pct': round(calculated_edge, 2)
            })
            
    return pd.DataFrame(trading_signals)

Comprehensive Python Engine: NOAA Bulk Data ParserThis production-grade script parses raw NOAA fixed-width bulk archive text data, converts the native metrics into exact Fahrenheit targets, reads the missing quality flags, and flags structural data gaps before writing them directly into your database.pythonimport pandas as pd
import numpy as np

def parse_noaa_bulk_file(file_path):
    """
    Parses fixed-width GHCN-Daily or raw NOAA bulk text files.
    Extracts elements, standardizes values, and isolates data quality flags.
    """
    parsed_records = []
    
    with open(file_path, 'r') as f:
        for line in f:
            # Standard GHCN text formats use fixed character column offsets
            station_id = line[0:11].strip()
            year = int(line[11:15])
            month = int(line[15:17])
            element = line[17:21].strip() # e.g., TMAX, TMIN, PRCP
            
            # We only isolate max temperatures for our Kalshi high contracts
            if element != "TMAX":
                continue
                
            # Loop through all 31 potential days in the text row
            char_pointer = 21
            for day in range(1, 32):
                # Each day consists of an 5-character value and 3 single-character flags
                val_str = line[char_pointer:char_pointer+5].strip()
                m_flag = line[char_pointer+5:char_pointer+6] # Measurement flag
                q_flag = line[char_pointer+6:char_pointer+7] # Quality flag
                s_flag = line[char_pointer+7:char_pointer+8] # Source flag
                
                char_pointer += 8
                
                # Check for missing data markers (-9999 or empty blanks)
                if not val_str or val_str == "-9999":
                    continue
                    
                    
                # NOAA bulk archives record temperatures in tenths of a degree Celsius
                temp_c = float(val_str) / 10.0
                temp_f = (temp_c * 9/5) + 32.0
                
                # Format a clean string target date
                try:
                    target_date = f"{year}-{month:02d}-{day:02d}"
                    pd.to_datetime(target_date) # Validate date existence
                except ValueError:
                    continue # Skip invalid calendar dates (e.g., Feb 30th)
                    
                parsed_records.append({
                    'station_code': station_id,
                    'target_date': target_date,
                    'latest_nws_tmax_f': round(temp_f, 1),
                    'quality_flag': q_flag.strip(),
                    'source_flag': s_flag.strip(),
                    'is_compromised': q_flag.strip() != "" # True if data failed any QC test
                })
                
    df = pd.DataFrame(parsed_records)
    return df

1. Hour-Angle Volatility Decay FunctionAdd this helper function to your pipeline. It compresses your standard deviation (σ) as the trading day approaches the late afternoon market settlement, forcing your probabilities to sharpen when retail market quotes lag.pythonfrom datetime import datetime
import pytz

def calculate_time_decayed_sigma(base_sigma, station_code):
    """
    Applies an exponential time-decay function to scale down trailing 30-day RMSE
    as the trading hours approach the standard peak heating settlement lock (4:00 PM local).
    """
    # Map station airports to their operational local timezones
    tz_mapping = {
        "KDEN": "America/Denver",
        "KMIA": "America/New_York",
        "KORD": "America/Chicago"
    }
    
    tz_name = tz_mapping.get(station_code, "America/New_York")
    local_tz = pytz.timezone(tz_name)
    now_local = datetime.now(local_tz)
    
    # Define standard daily trading evaluation window boundaries (7 AM to 4 PM)
    start_hour = 7
    settlement_hour = 16 
    
    current_hour_decimal = now_local.hour + (now_local.minute / 60.0)
    
    # Before the trading session starts, utilize full historic baseline variance
    if current_hour_decimal <= start_hour:
        return base_sigma
        
    # If trading continues past peak afternoon heating, compress to a knife-edge floor
    if current_hour_decimal >= settlement_hour:
        return 0.30
        
    # Calculate fraction of the daily heating cycle that has already concluded
    total_heating_duration = settlement_hour - start_hour
    hours_elapsed = current_hour_decimal - start_hour
    
    # Volatility decays along an exponential curve as raw uncertainty vanishes
    decay_factor = np.sqrt(1.0 - (hours_elapsed / total_heating_duration))
    
    # Maintain an operational uncertainty floor of 0.35°F to prevent math exceptions
    decayed_sigma = base_sigma * decay_factor
    return max(decayed_sigma, 0.35)
